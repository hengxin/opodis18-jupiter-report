\begin{frame}{}
  \begin{definition}[Eventual Convergence~\ncite{Ellis:SIGMOD89}]
    The lists are identical at all replicas \blue{at quiescence}, i.e.,
    all update operations have been executed at all replicas.
  \end{definition}

  \pause
  \vspace{0.50cm}
  \begin{definition}[Strong Eventual Consistency~\ncite{Shapiro:SSS11}]
    The lists are identical at all replicas whenever after executing \blue{the same set} of update operations.
  \end{definition}

  \pause
  \vspace{0.60cm}
  \centerline{\red{\large Specify little on {\it intermediate states} going through by replicas.}}
