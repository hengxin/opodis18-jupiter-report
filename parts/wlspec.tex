\begin{frame}{}
  \begin{definition}[Eventual Convergence~\ncite{Ellis:SIGMOD89}]
    The lists are identical at all replicas \blue{at quiescence}, \\
    i.e., all update operations have been executed at all replicas.
  \end{definition}

  \pause
  \vspace{0.50cm}
  \begin{definition}[Strong Eventual Consistency~\ncite{Shapiro:SSS11}]
    The lists are identical at the replicas whenever they have executed \\ \blue{the same set} of update operations.
  \end{definition}

  \pause
  \vspace{0.60cm}
  \centerline{\red{\large Specify little on \teal{\it intermediate states} going through by replicas.}}
\end{frame}

  \vspace{0.20cm}
  \begin{definition}[Weak List Specification \wlspec{}~\ncite{Attiya:PODC16}]
    Informally, \wlspec{} requires the ordering between \red{elements that are not deleted} to be consistent across the system.
  \end{definition}

  \vspace{0.60cm}
  \centerline{\red{\large Specify a global property \teal{\it on all states} across the system.}}
\end{frame}

  \end{definition}
